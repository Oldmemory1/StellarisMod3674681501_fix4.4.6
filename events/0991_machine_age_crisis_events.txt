# 风暴首次扩散后触发换空间站定时器，禁用后续其他堕落的风暴扩散
country_event = {
	id = crisis.8030
	hide_window = yes
	fire_only_once = yes
	is_triggered_only = yes

	immediate = {
		set_crisis_stage_2 = yes

		every_system = {
			limit = {
				has_star_flag = synth_queen_system
				NOT = {
					any_ship_in_system = {
						is_ship_class = shipclass_starbase
					}
				}
			}
			create_fleet = {
				name = "NAME_Synth_Queen_Starbase"
				effect = {
					set_owner = event_target:synth_queen_country_global
					create_ship = {
						name = random
						design = "NAME_Synth_Queen_Starbase"
					}
					set_location = {
						target = prev
					}
					while = {
						count = 2
						create_synth_queen_platform = yes
					}
				}
			}
		}

		every_country = {
			limit = {
				OR = {
					is_country_type = default
					is_country_type = fallen_empire
					is_country_type = awakened_fallen_empire
				}
			}
			country_event = { id = crisis.8039 days = 15 }
			if = {
				limit = {
					NOT = { is_country_type = default }
				}
				add_modifier = {
					modifier = beset_by_cetana
					days = -1
				}

			}
		}
		#除天灾出生国2格以内的其他堕落风暴
		country_event = { id = crisis.8042 days = 360 random = 50 } #Fallen Empire weakener

		country_event = { id = crisis.30009 days = 411 }
		#自定义事件循环删除空间站创建塞塔娜星垒
		country_event = { id = crisis.30000 days = 3 }
		set_global_flag = synth_queen_revenge_done
	}
}

	id = crisis.30000
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_system = {
			limit = { 
				#找轻蔑之幕星系

				owner = { is_country_type = synth_queen }
				# 筛选条件：星系带有“合成女王系统”标志（已占领），且非首都星系
				#has_star_flag = synth_queen_system  
				#NOT = { has_star_flag = synth_queen_capital }  
			}
			every_fleet_in_system = {
				# Destroy all stations 删除空间站否则不能创建塞塔娜星垒
				limit = {
					#是空间站
					is_mobile = no
					#空间站所有者是塞塔娜
					owner = { is_synth_queen_country_type = yes }
					#排除女王空间站模板
					NOT = { is_ship_size = big_starbase_synth_queen
							is_ship_size = starbase_synth_queen
					}
				}
				delete_fleet = this
			}
			
			#创建小星垒
			create_starbase = { 
				size = starbase_synth_queen
				owner = event_target:synth_queen_country_global
				effect = {
					while = {
						count = 2
						create_synth_queen_platform = yes
					}
				}
			}
		}
		country_event = { id = crisis.30000 days = 180 }
	}	
}
